cmake_minimum_required (VERSION 3.0)
project (ray-tracer)
add_executable(ray-tracer src/main.cpp)
target_compile_options (ray-tracer PRIVATE -std=c++17 -O3 -Wall -Wextra -Ofast)
target_compile_definitions (ray-tracer PRIVATE BOOST_ERROR_CODE_HEADER_ONLY)
target_link_libraries (ray-tracer -fopenmp)
find_package(OpenMP)
if (OPENMP_FOUND)
    set (CMAKE_C_FLAGS "${CMAKE_C_FLAGS} ${OpenMP_C_FLAGS}")
    set (CMAKE_CXX_FLAGS "${CMAKE_CXX_FLAGS} ${OpenMP_CXX_FLAGS}")
    set (CMAKE_EXE_LINKER_FLAGS "${CMAKE_EXE_LINKER_FLAGS} ${OpenMP_EXE_LINKER_FLAGS}")
endif()
